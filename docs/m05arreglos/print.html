A. Conceptos básicos

 const arr = ["mx", "my", "mu"]
 console.log(arr)
 console.log(arr.length)
 console.log(arr[0])
 console.log(arr[1])
 console.log(arr[2])
 arr[2] = "hi"
 const i = 0
 console.log(arr[i])
 arr[i] = "ho"
 console.log(arr[i])

B. Uso de contadores

const arr = ["mx", "my", "mu"]
for (let i = 0,
 len = arr.length;
 i < len;
 i++) {
 const s = arr[i]
 console.log(s)
}
console.log("adios")

Salida

mx
my
mu
adios

G. Arreglos de instancias de clases

class Clao {
 constructor(a) {
  this.a = a
 }
}
const arrI = [
 new Clao(-3),
 new Clao(4),
 new Clao(8)]
console.log(arrI)
for (const ref of arrI) {
 console.log(ref.a)
}

Salida

[ Clao { a: -3 }, Clao { a: 4 }, Clao { a: 8 } ]
-3
4
8
