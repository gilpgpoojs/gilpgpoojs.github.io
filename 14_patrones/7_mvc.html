<!DOCTYPE html>
<html lang="es">
  <head>
    <meta charset="UTF-8">
    <meta name="viewport"
      content="width=device-width,
       initial-scale=1.0">
    <title>Patrón MVC</title>
  </head>
  <script>
    function valida(cond,
      mensaje) {
      if (!cond) {
        throw new Error(mensaje);
      }
    }
    class Modelo {
      /** @param {number} a
       * @param {number} b */
      constructor(a, b) {
        this.a = a;
        this.b = b;
        /** @type {string} */
        this.salida = "";
      }
    }
    class Controlador {
      /** Realiza una división.
       * @param {Modelo} modelo
       * @throws {Error} si los
       *  parámetros no cumplen
       *  con las validaciones. */
      divide(modelo) {
        try {
          valida(!isNaN(modelo.a),
            "a debe ser número.");
          valida(!isNaN(modelo.b),
            "b debe ser número.");
          valida(modelo.b !== 0,
            "b no puede ser 0.");
          const resultado =
            modelo.a / modelo.b;
          modelo.salida =
            `${resultado}`;
        } catch (e) {
          console.error(e);
          modelo.salida =
            e.message;
        }
      }
    }
    class Vista
      extends HTMLElement {
      connectedCallback() {
        this.innerHTML = /*html*/
          `<p>
            <label>
              a <input id="a">
            </label>
          </p>
          <p>
            <label>
              b <input id="b">
            </label>
          </p>
          <p>
            <button type="submit"
             onclick="this.
                parentNode.
                parentNode.
                divide()">
              Saludar
            </button>
          </p>
          <p>
            <output id="salida">
            </output>
          </p>`;
        this.controlador =
          new Controlador();
      }
      divide() {
        let modelo = new Modelo(
          parseFloat(a.value),
          parseFloat(b.value));
        this.controlador.
          divide(modelo);
        salida.value =
          modelo.salida;
      }
    }
    customElements.define(
      "vista-divide", Vista);
  </script>
  <style>
    vista-divide {
      display: block;
    }
  </style>
  <body>
    <h1>Patrón MVC</h1>
    <vista-divide></vista-divide>
    <footer>
      <p>
        &copy; 2021
        Gilberto Pacheco Gallegos.
      </p>
    </footer>
  </body>
</html>
